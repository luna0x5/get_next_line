char    *ft_strjoin(char *s1, char *s2)
{
    size_t    s1len;
    size_t    s2len;
    char    *s;
    if (s1 == NULL && s2 == NULL)
        return (NULL);
    if (s1 == NULL)
        return (strdup(s2));
    if (s2 == NULL)
        return (strdup(s1));
    s1len = strlen(s1);
    s2len = strlen(s2);
    s = (char *)malloc((s1len + s2len + 1) * (sizeof(char)));
    if (s == NULL)
        return (NULL);
    strcpy(s, s1);
    strcpy(s + s1len, s2);
    return (s);
}
